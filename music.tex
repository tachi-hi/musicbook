SATB

\begin{lilypond}[quote,fragment,staffsize=16] 
\new StaffGroup <<
	\new Staff<<
		\clef soprano
		\set Staff.instrumentName = \markup{ Soprano }
		\relative c' {
			c'2 c b c
		}
	>>
	\new Staff<<
		\clef alto
		\set Staff.instrumentName = \markup{ Alto }
		\relative c' {
			e2 f d e
		}
	>>
	\new Staff <<
		\clef tenor
		\set Staff.instrumentName = \markup{ Tenor }
		\relative c{
			g'2 a g g
		}
	>>
	\new Staff<<
		\clef bass
		\set Staff.instrumentName = \markup{ Bass }
		\relative c{
			c2 f, g c
		}
	>>
>>
\end{lilypond}
