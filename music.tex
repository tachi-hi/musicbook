		}
	>>

		\clef alto
		\set Staff.instrumentName = \markup{ Alto }
		\relative c' {
			e2 f d e
		}
	>>
	\new Staff <<
		\clef tenor
		\set Staff.instrumentName = \markup{ Tenor }
		\relative c{
			g'2 a g g
		}
	>>
	\new Staff<<
		\clef bass
		\set Staff.instrumentName = \markup{ Bass }

		}
	>>
>>
\end{lilypond}
